class BankAccount{
    private long accountNumber;
    private String holderName;
    private int balance;
    
    
    BankAccount(long a,String name,int b){
        this.accountNumber=a;
        this.holderName=name;
        
        if (b>=0){
            this.balance=b;
        }
        else{
            System.out.println("Balance cannot be negative");
            this.balance=0;
        }
    }
    
    
    void deposit(int ammount){
        if (ammount>0){
            balance+=ammount;
            System.out.println("Ammount deposited successfully");
        }
        else{
            System.out.println("Enter proper ammount");
        }
    }
    
    void withdraw(int ammount){
        if (ammount>0 && ammount<=balance){
            balance-=ammount;
            System.out.println("Ammount withdrawed");
        }
        else{
            System.out.println("withdraw ammount excedes balance");
        }
    }
    
    int getBalance(){
        return balance;
    }
    
    void display(){
        System.out.println(accountNumber);
        System.out.println(holderName);
        System.out.println(balance);
    }
    
}

class Main {
    public static void main(String[] args) {
        BankAccount one=new BankAccount(12335656553322L,"SINGH",10000);
        one.deposit(2000);
        one.withdraw(1000);
        System.out.println(one.getBalance());
        one.display();
    }
}
